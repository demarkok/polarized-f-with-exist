\begin{restatable}[Positive Subtyping is Deterministic]{observation}{obsPosSubtDeterministic}
    \label{obs:pos-subt-deterministic}
    For fixed $[[Γ]]$, $[[Θ]]$, $[[uP]]$, and $[[iQ]]$,
    if $[[Γ ; Θ ⊨ uP ≥ iQ ⫤ SC]]$ and $[[Γ ; Θ ⊨ uP ≥ iQ ⫤ SC']]$
    then $[[SC]] = [[SC']]$.
\end{restatable}

\begin{restatable}[Soundness of the Positive Subtyping]{lemma}{lemPosSubtSoundness} \label{lemma:pos-subt-soundness}
    If $[[Γ ⊢ Θ]]$, $[[Γ ⊢ iQ]]$, $[[Γ ; dom(Θ) ⊢  uP]]$, and 
    $[[Γ ; Θ ⊨ uP ≥ iQ ⫤ SC]]$,
    then $[[Θ ⊢ SC : uv uP]]$ and
    for any normalized $[[uσ]]$ such that $[[ Θ ⊢ uσ : SC ]]$,
    $[[ Γ ⊢ [uσ]uP ≥ iQ ]]$.
\end{restatable}

\begin{restatable}[Completeness of the Positive Subtyping]{lemma}{lemPosSubtCompleteness} \label{lemma:pos-subt-completeness}
    Suppose that $[[Γ ⊢ Θ]]$, $[[Γ ⊢ iQ]]$ and $[[Γ ; dom(Θ) ⊢  uP]]$.
    Then for any $[[Θ ⊢ uσ : uv uP]]$ such that $[[ Γ ⊢ [uσ]uP ≥ iQ ]]$,
    there exists $[[Γ; Θ ⊨ uP ≥ iQ ⫤ SC]]$ and moreover, $[[ Θ ⊢ uσ : SC ]]$.
\end{restatable}
